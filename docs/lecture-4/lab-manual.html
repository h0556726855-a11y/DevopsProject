3

Write the four Terraform files

Create each file exactly as shown. The four files together provision a VPC, IAM roles, the EKS control plane, and a single node group.

terraform/versions.tf
terraform {
  required_version = ">= 1.6"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }
}

provider "aws" {
  region = var.aws_region
}
terraform/variables.tf
variable "aws_region" {
  description = "AWS region for all resources"
  type        = string
  default     = "eu-west-1"
}

variable "cluster_name" {
  description = "Name of the EKS cluster"
  type        = string
  default     = "shoplist"
}

variable "cluster_version" {
  description = "Kubernetes version for EKS"
  type        = string
  default     = "1.29"
}

variable "node_instance_type" {
  description = "EC2 instance type for the worker node"
  type        = string
  default     = "t3.micro"
}
terraform/networking.tf
# Fetch the list of available AZs in the selected region
data "aws_availability_zones" "available" {
  state = "available"
}

# VPC
resource "aws_vpc" "main" {
  cidr_block           = "10.0.0.0/16"
  enable_dns_hostnames = true
  enable_dns_support   = true

  tags = {
    Name                                          = "shoplist-vpc"
    "kubernetes.io/cluster/${var.cluster_name}" = "shared"
  }
}

# Internet Gateway
resource "aws_internet_gateway" "main" {
  vpc_id = aws_vpc.main.id
  tags = { Name = "shoplist-igw" }
}

# Two public subnets — one per AZ (required by EKS)
resource "aws_subnet" "public" {
  count                   = 2
  vpc_id                  = aws_vpc.main.id
  cidr_block              = cidrsubnet("10.0.0.0/16", 8, count.index)
  availability_zone       = data.aws_availability_zones.available.names[count.index]
  map_public_ip_on_launch = true

  tags = {
    Name                                          = "shoplist-public-${count.index}"
    "kubernetes.io/cluster/${var.cluster_name}" = "shared"
    "kubernetes.io/role/elb"                      = "1"
  }
}

# Route table — all outbound traffic via IGW
resource "aws_route_table" "public" {
  vpc_id = aws_vpc.main.id
  route {
    cidr_block = "0.0.0.0/0"
    gateway_id = aws_internet_gateway.main.id
  }
  tags = { Name = "shoplist-public-rt" }
}

resource "aws_route_table_association" "public" {
  count          = 2
  subnet_id      = aws_subnet.public[count.index].id
  route_table_id = aws_route_table.public.id
}
terraform/eks.tf
# IAM role for the EKS control plane
resource "aws_iam_role" "eks_cluster" {
  name = "shoplist-eks-cluster-role"
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action    = "sts:AssumeRole"
      Effect    = "Allow"
      Principal = { Service = "eks.amazonaws.com" }
    }]
  })
}

resource "aws_iam_role_policy_attachment" "eks_cluster_policy" {
  role       = aws_iam_role.eks_cluster.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonEKSClusterPolicy"
}

# IAM role for the worker node group
resource "aws_iam_role" "eks_nodes" {
  name = "shoplist-eks-node-role"
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action    = "sts:AssumeRole"
      Effect    = "Allow"
      Principal = { Service = "ec2.amazonaws.com" }
    }]
  })
}

resource "aws_iam_role_policy_attachment" "eks_worker_node_policy" {
  role       = aws_iam_role.eks_nodes.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonEKSWorkerNodePolicy"
}

resource "aws_iam_role_policy_attachment" "eks_cni_policy" {
  role       = aws_iam_role.eks_nodes.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonEKS_CNI_Policy"
}

resource "aws_iam_role_policy_attachment" "eks_ecr_readonly" {
  role       = aws_iam_role.eks_nodes.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonEC2ContainerRegistryReadOnly"
}

# EKS control plane
resource "aws_eks_cluster" "shoplist" {
  name     = var.cluster_name
  role_arn = aws_iam_role.eks_cluster.arn
  version  = var.cluster_version

  vpc_config {
    subnet_ids              = aws_subnet.public[*].id
    endpoint_public_access  = true
    endpoint_private_access = false
  }

  depends_on = [aws_iam_role_policy_attachment.eks_cluster_policy]
}

# Managed node group — one t3.micro node
resource "aws_eks_node_group" "shoplist" {
  cluster_name    = aws_eks_cluster.shoplist.name
  node_group_name = "shoplist-nodes"
  node_role_arn   = aws_iam_role.eks_nodes.arn
  subnet_ids      = aws_subnet.public[*].id
  instance_types  = [var.node_instance_type]

  scaling_config {
    desired_size = 1
    max_size     = 1
    min_size     = 1
  }

  depends_on = [
    aws_iam_role_policy_attachment.eks_worker_node_policy,
    aws_iam_role_policy_attachment.eks_cni_policy,
    aws_iam_role_policy_attachment.eks_ecr_readonly,
  ]
}

# Outputs
output "cluster_endpoint" {
  value = aws_eks_cluster.shoplist.endpoint
}

output "cluster_name" {
  value = aws_eks_cluster.shoplist.name
}

2

Update Kubernetes Manifests

Steps 6–8 ~15 min kubectl YAML
6

Edit kubernetes/frontend-service.yml to use LoadBalancer

The frontend service type must be LoadBalancer so AWS automatically provisions a Classic ELB. Open the file and change type: ClusterIP to type: LoadBalancer.

kubernetes/frontend-service.yml — after edit
apiVersion: v1
kind: Service
metadata:
  name: frontend
  namespace: shoplist
spec:
  type: LoadBalancer     # changed from ClusterIP
  selector:
    app: frontend
  ports:
    - port: 80
      targetPort: 80
7

Confirm __IMAGE_TAG__ placeholder in backend deployment

Open kubernetes/backend-deployment.yml and confirm the image line uses __IMAGE_TAG__ as the tag. The CI pipeline will substitute this with the real commit SHA at deploy time.

kubernetes/backend-deployment.yml — relevant line
          image: registry.gitlab.com/YOUR_GROUP/devops-tikshuv-project/backend:__IMAGE_TAG__
Replace YOUR_GROUP with your actual GitLab group/namespace. For example: registry.gitlab.com/acme-devops/devops-tikshuv-project/backend:__IMAGE_TAG__
8

Confirm __IMAGE_TAG__ placeholder in frontend deployment

Open kubernetes/frontend-deployment.yml and confirm the same pattern.

kubernetes/frontend-deployment.yml — relevant line
          image: registry.gitlab.com/YOUR_GROUP/devops-tikshuv-project/frontend:__IMAGE_TAG__

Both files should now have __IMAGE_TAG__ as their image tag. Do not put a real SHA here — the sed command in the pipeline replaces it.

Configure the CI/CD Pipeline

Steps 9–12 ~20 min GitLab CI IAM
9

Add four CI/CD variables in GitLab (all masked)

Navigate to your GitLab project → Settings → CI/CD → Variables. Add each variable below. Tick Masked for all four to prevent values from appearing in job logs.

Variable Name Value Masked Notes
AWS_ACCESS_KEY_ID Your IAM access key ID Yes From IAM → Security Credentials
AWS_SECRET_ACCESS_KEY Your IAM secret access key Yes Only shown once at key creation
AWS_REGION eu-west-1 Yes Must match the Terraform region
CLUSTER_NAME shoplist Yes Must match var.cluster_name in Terraform
10

Add the deploy stage to .gitlab-ci.yml

Your current .gitlab-ci.yml has build and verify stages. Add a deploy stage and the job below. The alpine/k8s:1.29.2 image bundles both kubectl and the AWS CLI.

.gitlab-ci.yml — add to top-level stages list
stages:
  - build
  - verify
  - deploy          # add this line
.gitlab-ci.yml — add the deploy job
deploy:
  stage: deploy
  image: alpine/k8s:1.29.2
  script:
    - aws eks update-kubeconfig --name $CLUSTER_NAME --region $AWS_REGION
    - kubectl apply -f kubernetes/namespace.yml
    - kubectl apply -f kubernetes/postgres-secret.yml
    - kubectl apply -f kubernetes/postgres-configmap.yml
    - kubectl apply -f kubernetes/postgres-pvc.yml
    - kubectl apply -f kubernetes/postgres-deployment.yml
    - kubectl apply -f kubernetes/postgres-service.yml
    - sed "s|__IMAGE_TAG__|$CI_COMMIT_SHORT_SHA|g" kubernetes/backend-deployment.yml | kubectl apply -f -
    - kubectl apply -f kubernetes/backend-service.yml
    - sed "s|__IMAGE_TAG__|$CI_COMMIT_SHORT_SHA|g" kubernetes/frontend-deployment.yml | kubectl apply -f -
    - kubectl apply -f kubernetes/frontend-service.yml
    - kubectl rollout status deployment/backend -n shoplist --timeout=120s
    - kubectl rollout status deployment/frontend -n shoplist --timeout=120s
  needs:
    - verify-registry
  rules:
    - if: $CI_COMMIT_BRANCH == $CI_DEFAULT_BRANCH
The sed command on lines 9 and 11 replaces the literal string __IMAGE_TAG__ in the manifest with the real $CI_COMMIT_SHORT_SHA before piping to kubectl apply. The manifest files on disk are never modified.

Verify the running image SHA matches the last commit

Confirm the image tag on the live backend pod matches the git commit SHA that triggered the pipeline.

Commands
# Get the commit SHA the pipeline ran on
git rev-parse --short HEAD

# Get the image tag currently running in the cluster
kubectl get pod -n shoplist \

5

DESTROY — Tear Down All Resources

Step 17 Required ~15 min
This step is not optional. Every minute you leave the cluster running costs money. Run these three sub-steps before you close your laptop.
17

Destroy the cluster and all AWS resources

Sub-step A: Delete the Kubernetes namespace first. This removes the LoadBalancer service, which instructs AWS to delete the Classic ELB. If you skip this, Terraform destroy may fail because the ELB is still attached to the VPC.

Sub-step A — delete namespace
kubectl delete namespace shoplist

# Wait for the namespace to terminate (ELB deletion takes ~60s)
kubectl wait --for=delete namespace/shoplist --timeout=120s
Expected output
namespace "shoplist" deleted

Sub-step B: Run terraform destroy from the terraform/ directory. This deletes the node group, EKS control plane, VPC, subnets, route tables, IAM roles, and internet gateway.

Sub-step B — terraform destroy
cd terraform
terraform destroy

Type yes when prompted. This takes 8–12 minutes.

Expected: final output
Destroy complete! Resources: 14 destroyed.

Sub-step C: Verify in the AWS Console that no resources remain.

Sub-step C — console verification checklist
AWS Console Section What you should see
EKS → Clusters No clusters listed (or "No clusters")
EC2 → Instances No running instances (t3.micro terminated)
EC2 → Load Balancers No load balancers
VPC → Your VPCs Only the default VPC; shoplist-vpc is gone
If terraform destroy fails with a dependency error (e.g., security group still attached to an ENI), wait 2 minutes and run terraform destroy again. The Classic ELB deletion sometimes takes longer than expected to release network interfaces.

Troubleshooting Reference

Quick Reference
Symptom Likely Cause Fix
Node shows NotReady VPC CNI addon still initialising Wait 90 seconds, re-run kubectl get nodes
EXTERNAL-IP stuck at <pending> ELB provisioning takes 60–90 s Run kubectl get svc -n shoplist -w and wait for the hostname to appear
terraform apply fails with "OptInRequired" EKS not enabled for your AWS account region Check you are using eu-west-1; confirm EKS is available in your account via the Console
deploy job fails: error: exec plugin IAM credentials not passed to the job Verify AWS_ACCESS_KEY_ID and AWS_SECRET_ACCESS_KEY are set and masked in GitLab CI/CD Variables
ImagePullBackOff on backend or frontend pod Image tag does not exist in registry, or YOUR_GROUP placeholder not replaced Check the image URL in the deployment: kubectl describe pod -n shoplist -l app=backend
curl /api/health returns connection refused Backend pod not yet Ready kubectl get pods -n shoplist — wait for all pods to be Running
terraform destroy fails with ENI dependency error Classic ELB still releasing network interfaces Wait 2 minutes and re-run terraform destroy. Also ensure you ran Sub-step A first
Pipeline deploy stage skipped Pushed to a branch other than main The deploy job has if: $CI_COMMIT_BRANCH == $CI_DEFAULT_BRANCH. Push to main to trigger it
